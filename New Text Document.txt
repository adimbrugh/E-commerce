SELECT categories.name,products.id as new_product_id, products.name, products.price , product_colors.id as new_color_id, product_colors.color, product_color_images.image, product_sizes.size FROM categories, products ,product_colors, product_color_images, product_sizes WHERE categories.id = products.category_id AND products.id=product_colors.product_id AND product_colors.id = product_color_images.product_color_id AND products.id = product_sizes.product_id;
